## Color Schemes

### Option 1: Status-based

```typescript
const getNodeColor = (node: GraphNode) => {
  switch (node.type) {
    case 'hub':    return '#3b82f6'; // Blue - Important
    case 'orphan': return '#ef4444'; // Red - Needs attention
    default:       return '#22c55e'; // Green - Normal
  }
};
```

### Option 2: Tag-based

```typescript
const tagColors = {
  'project': '#3b82f6',
  'personal': '#22c55e',
  'work': '#f59e0b',
  'idea': '#8b5cf6',
};
```

### Option 3: Connection Density

```typescript
const getNodeColor = (connections: number) => {
  if (connections > 10) return '#dc2626'; // Deep red
  if (connections > 5)  return '#f59e0b'; // Orange
  if (connections > 2)  return '#3b82f6'; // Blue
  return '#22c55e'; // Green
};
```
